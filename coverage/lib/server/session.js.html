'use strict';
 
const debug = require('debug')('Durga:Session');
const debugVerbose = require('debug')('verbose:Durga:Session');
const UUID = require('uuid/v1');
 
 
module.exports = class DurgaSession {
 
	constructor(server) {
		debug('CREATE');
 
		this.server = server;
		this.id = UUID();
	}
 
	log(tags, data) {
		this.server.log(['session', ...tags], data);
		return this;
	}
 
	destroy() {
		debug('DESTROY');
		return this.server.hooks.run('session.destroy');
	}
 
 
 
	listen(handler) {
		this._listener = function() {
			debugVerbose('Send', ...arguments);
			handler(...arguments);
		};
	}
 
 
 
	sendRaw(e) {
		if(!this._listener) {
			throw new Error('Listener not attached');
		}
		this._listener(e);
	}
 
	send(composerName, attrs) {
		let msg = this.server.protocol.compose(composerName, attrs);
		return this.sendRaw(msg);
	}
 
 
	dispatch(event) {
		let handler = this.server.protocol.match(event);
 
		if(!handler) {
			this.log(['error', 'dispatch'], { event });
			return Promise.reject(new Error('Handler not found'));
		}
 
		return this.server.runHandler(handler, this, event);
	}
 
};
