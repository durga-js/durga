'use strict';
 
const Collection = require('./collection');
 
 
class CollectionsError extends Error {}
 
module.exports = class Proxies {
 
	constructor(server, proxies) {
		this.server = server;
		this.proxies = proxies;
 
		this._collections = {};
	}
 
	getSet(name, proxy, config) {
 
		// set
		if(proxy) {
			if(this._collections.hasOwnProperty(name)) {
				throw new CollectionsError(`Collection '${name}' already defined.`);
			}
 
 
			proxy = this.proxies.getSet(proxy);
			return (this._collections[name] = new Collection(name, proxy, config));
 
		// get
		} else {
			if(this._collections.hasOwnProperty(name)) {
				return this._collections[name];
			} else {
				throw new CollectionsError(`Collection '${name}' not defined`);
			}
		}
 
	}
 
};
