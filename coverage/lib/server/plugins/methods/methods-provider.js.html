'use strict';
 
 
class MethodsProviderError extends Error {}
 
class ReplyEmitter {
 
}
 
module.exports = class Proxies {
 
	constructor(server) {
		this.server = server;
 
		this._methods = {};
	}
 
	getSet(name, config) {
 
		// set
		if(config) {
			if(this._methods.hasOwnProperty(name)) {
				throw new MethodsProviderError(`Method '${name}' already defined.`);
			}
 
			return (this._methods[name] = this._createHandler(config));
 
		// get
		} else {
			if(this._methods.hasOwnProperty(name)) {
				return this._methods[name];
			} else {
				throw new MethodsProviderError(`Method '${name}' not found`);
			}
		}
 
	}
 
	get(name) {
		return this.getSet(name);
	}
 
	set(name, config) {
		return this.getSet(name, config);
	}
 
	get TYPE() {
		return 'method';
	}
 
 
	_createHandler(config) {
 
 
		return this.server.createHandler(this.TYPE, config, (err, ctx, handler) => {
 
			if(err) {
				return ctx.session.send('methodReply', {
					eid: ctx.event.eid,
					error: true,
					payload: err
				});
			}
 
			return Promise.resolve(handler(ctx))
				.then(res => ctx.session.send('methodReply', {
					payload: res,
					eid: ctx.event.eid
				}));
 
		});
 
	}
 
};
 
module.exports.MethodsProviderError = MethodsProviderError;
