'use strict';
 
const MethodsProvider = require('./methods-provider');
 
module.exports = function(server, options) {
 
 
	let methods = new MethodsProvider(server);
 
	server.decorate('server', 'method', methods.getSet.bind(methods));
 
	server.protocol
		.registerMatcher(event => {
			if(event.type === 'method' && event.method && event.eid) {
				return methods.get(event.method);
			}
		})
		.registerComposer('methodReply', ({ eid, payload, error }) => ({
			type: 'method',
			eid,
			error,
			payload
		}))
 
 
	server.hook('preHandler', (ctx) => {
 
		if(ctx.handler.type === methods.TYPE) {
			ctx.payload = ctx.event.payload;
		}
 
	});
 
};
